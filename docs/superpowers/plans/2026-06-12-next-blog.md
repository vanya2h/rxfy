# Next.js Blog Example Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a Next.js 15 App Router blog example in `examples/next-blog/` that demonstrates rxfy's parameterized state, normalized model stores, streaming SSR via `HydrationStream`, and local mutations — with posts, users, and comments.

**Architecture:** Two routes (`/` posts list, `/posts/[id]` post detail) backed by in-memory data. All rxfy hook usage lives in `"use client"` components; Server Components only resolve route params. `HydrationStream` from `rxfy-react/next` streams normalized state deltas to the client via `useServerInsertedHTML`.

**Tech Stack:** Next.js 15, React 19, rxfy (workspace:*), rxfy-react (workspace:*), RxJS 7, Zod 3, TypeScript 5

## Task 3: rxfy State Layer

**Files:**
- Create: `examples/next-blog/src/blog.ts`

- [ ] **Step 3.1: Write `examples/next-blog/src/blog.ts`**

Defines Zod schemas, rxfy models, parameterized state descriptors, fetchers, and the `createComment` mutation helper.

```ts
import { array, createModel, defineState, single } from "rxfy";
import { z } from "zod";
import { db } from "./db.js";

// ── Schemas & types ────────────────────────────────────────────────────────────

export const UserSchema = z.object({
  id: z.string(),
  name: z.string(),
  email: z.string(),
});

export const PostSchema = z.object({
  id: z.string(),
  userId: z.string(),
  title: z.string(),
  body: z.string(),
});

export const CommentSchema = z.object({
  id: z.string(),
  postId: z.string(),
  name: z.string(),
  body: z.string(),
});

export type User = z.infer<typeof UserSchema>;
export type Post = z.infer<typeof PostSchema>;
export type Comment = z.infer<typeof CommentSchema>;

// ── rxfy models ────────────────────────────────────────────────────────────────

export const userModel = createModel(UserSchema, { getKey: (x) => x.id, name: "user" });
export const postModel = createModel(PostSchema, { getKey: (x) => x.id, name: "post" });
export const commentModel = createModel(CommentSchema, { getKey: (x) => x.id, name: "comment" });

// ── State definitions ──────────────────────────────────────────────────────────

export const postsState = defineState({
  key: "posts",
  params: z.object({}),
  model: { posts: array(postModel) },
  mutations: {},
});

export const postDetailState = defineState({
  key: "post-detail",
  params: z.object({ postId: z.string() }),
  model: {
    post: single(postModel),
    author: single(userModel),
    comments: array(commentModel),
  },
  mutations: {
    addComment: (prev, comment: Comment) => ({
      ...prev,
      comments: [...prev.comments, comment],
    }),
  },
});

// ── Fetchers ───────────────────────────────────────────────────────────────────

export async function fetchPosts(_: Record<never, never>, signal: AbortSignal): Promise<{ posts: Post[] }> {
  await delay(400, signal);
  return { posts: db.posts };
}

export async function fetchPostDetail(
  { postId }: { postId: string },
  signal: AbortSignal,
): Promise<{ post: Post; author: User; comments: Comment[] }> {
  await delay(400, signal);
  const post = db.posts.find((p) => p.id === postId);
  if (!post) throw new Error(`Post "${postId}" not found`);
  const author = db.users.find((u) => u.id === post.userId);
  if (!author) throw new Error(`Author "${post.userId}" not found`);
  const comments = db.comments.filter((c) => c.postId === postId);
  return { post, author, comments };
}

// ── Mutations ──────────────────────────────────────────────────────────────────

export function createComment(postId: string, name: string, body: string): Comment {
  const comment: Comment = { id: String(db.nextCommentId++), postId, name, body };
  db.comments = [...db.comments, comment];
  return comment;
}

// ── Helpers ────────────────────────────────────────────────────────────────────

function delay(ms: number, signal: AbortSignal): Promise<void> {
  return new Promise((resolve, reject) => {
    const id = setTimeout(resolve, ms);
    signal.addEventListener("abort", () => {
      clearTimeout(id);
      reject(signal.reason);
    });
  });
}
```

## Task 5: Posts List

- [ ] **Step 5.1: Write `examples/next-blog/src/components/PostList.tsx`**

`PostList` subscribes to `postsState` and renders a list. Each `PostItem` subscribes reactively to its entity via the model store, so a post update anywhere propagates without re-fetching the list.

```tsx
"use client";

import Link from "next/link";
import { useMemo } from "react";
import { Pending } from "rxfy-react";
import { useModelStore } from "rxfy-react";
import { fetchPosts, postModel, postsState, type Post } from "../blog.js";
import { useStateData } from "rxfy-react";

export default function PostList() {
  const { data$ } = useStateData(postsState, fetchPosts, {});

  return (
    <div>
      <h1>Blog</h1>
      <Pending
        value$={data$}
        pending={<p className="status">Loading posts…</p>}
        rejected={({ onReload }) => (
          <p className="status error">
            Failed to load.{" "}
            <button onClick={onReload}>Retry</button>
          </p>
        )}
      >
        {({ posts }) =>
          posts.length === 0 ? (
            <p className="status">No posts yet.</p>
          ) : (
            <ul className="post-list">
              {posts.map((id) => (
                <PostItem key={id} id={id} />
              ))}
            </ul>
          )
        }
      </Pending>
    </div>
  );
}

function PostItem({ id }: { id: string }) {
  const store = useModelStore(postModel);
  const post$ = useMemo(() => store.get(id), [store, id]);

  return (
    <Pending value$={post$}>
      {(post) => <PostItemContent post={post} />}
    </Pending>
  );
}

function PostItemContent({ post }: { post: Post }) {
  return (
    <li>
      <Link href={`/posts/${post.id}`}>
        <h2>{post.title}</h2>
      </Link>
      <p className="post-excerpt">{post.body.slice(0, 120)}…</p>
    </li>
  );
}
```

## Task 6: Add Comment Form

**Files:**
- Create: `examples/next-blog/src/components/AddCommentForm.tsx`

- [ ] **Step 6.1: Write `examples/next-blog/src/components/AddCommentForm.tsx`**

Controlled form with `name` and `body` fields. On submit: writes to the in-memory db via `createComment`, then fires the `addComment` mutation from the parent — the comment appears instantly in the UI with no re-fetch.

```tsx
"use client";

import { useState } from "react";
import { createComment, type Comment } from "../blog.js";

type AddCommentFormProps = {
  postId: string;
  onAdd: (comment: Comment) => void;
};

export default function AddCommentForm({ postId, onAdd }: AddCommentFormProps) {
  const [name, setName] = useState("");
  const [body, setBody] = useState("");

  const handleSubmit = (e: React.FormEvent<HTMLFormElement>) => {
    e.preventDefault();
    const trimmedName = name.trim();
    const trimmedBody = body.trim();
    if (!trimmedName || !trimmedBody) return;
    const comment = createComment(postId, trimmedName, trimmedBody);
    onAdd(comment);
    setName("");
    setBody("");
  };

  return (
    <form className="add-comment" onSubmit={handleSubmit}>
      <h3>Leave a comment</h3>
      <div>
        <label htmlFor="comment-name">Name</label>
        <input
          id="comment-name"
          value={name}
          onChange={(e) => setName(e.target.value)}
          placeholder="Your name"
          autoComplete="name"
        />
      </div>
      <div>
        <label htmlFor="comment-body">Comment</label>
        <textarea
          id="comment-body"
          value={body}
          onChange={(e) => setBody(e.target.value)}
          placeholder="Write your comment…"
        />
      </div>
      <button type="submit">Post comment</button>
    </form>
  );
}
```

- [ ] **Step 7.1: Write `examples/next-blog/src/components/PostDetail.tsx`**

`PostDetail` subscribes to `postDetailState` parameterized by `postId`. The normalized `data$` emits `{ post: string, author: string, comments: string[] }` (IDs). `PostBody` resolves the entities from model stores and uses `combineLatest` to merge them into a single subscription.

```tsx
"use client";

import Link from "next/link";
import { useMemo } from "react";
import { Pending } from "rxfy-react";
import { useModelStore, useStateData } from "rxfy-react";
import { combineLatest } from "rxjs";
import {
  commentModel,
  fetchPostDetail,
  postDetailState,
  postModel,
  userModel,
  type Comment,
  type Post,
  type User,
} from "../blog.js";
import AddCommentForm from "./AddCommentForm.js";

type DetailIds = { post: string; author: string; comments: string[] };

export default function PostDetail({ postId }: { postId: string }) {
  const params = useMemo(() => ({ postId }), [postId]);
  const { data$, mutations } = useStateData(postDetailState, fetchPostDetail, params);

  return (
    <div>
      <Link className="back-link" href="/">← All posts</Link>
      <Pending
        value$={data$}
        pending={<p className="status">Loading post…</p>}
        rejected={({ onReload }) => (
          <p className="status error">
            Failed to load.{" "}
            <button onClick={onReload}>Retry</button>
          </p>
        )}
      >
        {(ids) => (
          <PostBody
            ids={ids}
            postId={postId}
            onAddComment={(comment) => mutations.addComment(comment)}
          />
        )}
      </Pending>
    </div>
  );
}

function PostBody({
  ids,
  postId,
  onAddComment,
}: {
  ids: DetailIds;
  postId: string;
  onAddComment: (comment: Comment) => void;
}) {
  const postStore = useModelStore(postModel);
  const userStore = useModelStore(userModel);

  const post$ = useMemo(() => postStore.get(ids.post), [postStore, ids.post]);
  const author$ = useMemo(() => userStore.get(ids.author), [userStore, ids.author]);
  const combined$ = useMemo(() => combineLatest({ post: post$, author: author$ }), [post$, author$]);

  return (
    <Pending value$={combined$}>
      {({ post, author }) => (
        <PostArticle
          post={post}
          author={author}
          commentIds={ids.comments}
          postId={postId}
          onAddComment={onAddComment}
        />
      )}
    </Pending>
  );
}

function PostArticle({
  post,
  author,
  commentIds,
  postId,
  onAddComment,
}: {
  post: Post;
  author: User;
  commentIds: string[];
  postId: string;
  onAddComment: (comment: Comment) => void;
}) {
  return (
    <article>
      <h1>{post.title}</h1>
      <div className="author-block">
        {author.name} · <a href={`mailto:${author.email}`}>{author.email}</a>
      </div>
      <p className="post-body">{post.body}</p>
      <section>
        <h2>Comments ({commentIds.length})</h2>
        {commentIds.length === 0 ? (
          <p className="status">No comments yet. Be the first!</p>
        ) : (
          <ul className="comment-list">
            {commentIds.map((id) => (
              <CommentItem key={id} id={id} />
            ))}
          </ul>
        )}
        <AddCommentForm postId={postId} onAdd={onAddComment} />
      </section>
    </article>
  );
}

function CommentItem({ id }: { id: string }) {
  const store = useModelStore(commentModel);
  const comment$ = useMemo(() => store.get(id), [store, id]);

  return (
    <Pending value$={comment$}>
      {(comment) => (
        <li>
          <p className="comment-author">{comment.name}</p>
          <p className="comment-body">{comment.body}</p>
        </li>
      )}
    </Pending>
  );
}
```
